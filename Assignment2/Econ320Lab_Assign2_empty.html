# Define the CPI values for 1992 and 2008
cpi_92 = 140.3
cpi_08 = 215.2

# Calculate the inflation adjustment factor
inflator = cpi_08 / cpi_92

# Apply inflation adjustment to the 'ahe' column
df['[ahe_adj]'] = df.apply(lambda row: row['ahe'] * inflator
